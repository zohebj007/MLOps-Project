1)sudo apt update -y
2)sudo apt install -y wget unzip gnupg software-properties-common

3)wget -O- https://apt.releases.hashicorp.com/gpg | \
gpg --dearmor | \
sudo tee /usr/share/keyrings/hashicorp-archive-keyring.gpg

4)echo "deb [signed-by=/usr/share/keyrings/hashicorp-archive-keyring.gpg] \
https://apt.releases.hashicorp.com $(lsb_release -cs) main" | \
sudo tee /etc/apt/sources.list.d/hashicorp.list

5)sudo apt update -y
6)sudo apt install terraform -y

7)terraform -v

# aws cli on ubuntu

1)curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "awscliv2.zip"

2)unzip awscliv2.zip

3) if unzip is missing sudo apt install unzip -y

4)sudo ./aws/install

5)echo 'export PATH=/usr/local/bin:$PATH' >> ~/.bashrc
source ~/.bashrc

6) aws --version
